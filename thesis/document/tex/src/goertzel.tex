N = len(samples)
k = int((N*freq)/sample_rate)
w = 2.0 * np.pi / N * k
cosine = np.cos(w)
coeff = 2 * cosine
q0 = 0.0
q1 = 0.0
q2 = 0.0
for num in range(N):
	q0 = coeff * q1 - q2 + samples[num]
	q2 = q1
	q1 = q0

result = q1*q1 + q2*q2 - q1*q2*coeff
